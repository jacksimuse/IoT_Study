{

}

## 2차원 배열의 합(백준 2467번 문제)
```C
/*
백준 2167 : 2차원 배열의 합

예제로 나온 입력을 분석해보자면

2 3       // 배열의 크기 입력
1 2 4     // 배열의 요소 입력
8 16 32
3         // 합 몇번 할껀지 테스트케이스 입력
1 1 2 3   // 배열의 요소들의 합(3가지) : 63
1 2 1 2   // 2
1 3 2 3   36


int arr[2][3] = { {1, 2, 4}, {8, 16, 32}};
(1,1) ~ (2,3) 들의 합은 모든 배열 요소의 합이므로 63 출력
(1,2) ~ (1,2) 합은 지 자신밖에없으므로 arr[0][1] 값인 2 출력
(1,3) ~ (2,3) 합은 36


2차원 배열이 주어졌을때 (a,b), (x,y) 위치에 저장되있는 수들의 합을 구하는 문제.
1. 배열의 크기를 처음에(300, 300) 크게 잡고, 입력받은 만큼의 숫자를 for문을통해 배열의 요소를 입력받음.
2. testCase로 입력받은 만큼 합을 구할것임(for문의 횟수가될것임)
3. (a, b) (x, y)를 입력받고 2중 for문으로 배열의 요소의 합을 total에 저장한 후 출력
4. ★좌표입력받고 for문할때 배열은 0부터 시작하지만 좌표는 1부터 시작하므로 초기화식이나 조건문에 -1을 해주는 것을 기억★

*/

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() 
{
	int a, b, x, y;                   // 입력받을 좌표(a,b), (x,y)
	int arr[300][300] = { {0}, {0} }; // 입력받는 배열의 크기가 최대 300이므로 300으로 크기지정
	int N, M;                         // 배열의 크기를 입력받을 변수
	int i, j, k;                      // for문때 반복할때 쓸 변수
	int testCase;                     // 몇번 합을 구할지 쓸때쓰는 변수
	
	scanf("%d %d", &N, &M);           // 입력받는 N, M이 배열의 크기가 됨
	
	for (i = 0; i < N; i++) 
	{
		for (j = 0; j < M; j++) 
		{
			scanf("%d", &arr[i][j]);           // 배열이지만 정수일때는 꼭 &를 붙여줘야함. 아님 에러뜸..
		}
	}
	
	scanf("%d", &testCase);
	for (i = 0; i < testCase; i++) 
	{
		int total = 0;
		scanf("%d %d %d %d", &a, &b, &x, &y);     // (a,b), (x,y) 좌표 입력
		for (j = a - 1; j < x; j++)               // 입력한 좌표만큼 for문을 돌림
		{
			for (k = b - 1; k < y; k++) 
			{
				total += arr[j][k];       // 좌표들의 합을 total에 저장
			}
		}
		printf("%d\n", total);  // total 출력
	}
	return 0;
}
```
