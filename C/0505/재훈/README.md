```C
// 1. 문자열 입력받기  2. r번 반복하기 3. 반복한 문자열 출력

// 의사 코드 작성
	// 문자열을 배열 생성, 테스트 수, 반복할 수 생성
	// 값 받기

	//반복문 돌리기
	//글자길이만큼 돌리되 글자 하나 고정시키고 반복수 만큼 돌면 다음 글자로 넘어감


#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>

int main()
{
	char S[20] = { 0 }; // 20자 짜리 문자열 받기
	int T = 0, R = 0; // T 테스트 케이스, 문자하나 반복할 R

	scanf("%d", &T); // 전체 테스트 케이스를 먼저 받기

	for (int i = 0; i < T; i++) // 테스트 케이스만큼 돌린다는 반복문
	{
		scanf("%d %s", &R, S); // 문자하나 반복할 R과 문자열 S 받기

		for (int j = 0; j < strlen(S); j++) // 문자열S 길이만큼 반복해준다
		{
			for (int k = 0; k < R; k++) // 문자 하나를 R만큼 반복
			{
				printf("%c", S[j]); // 해당 순서에 있는 문자를 출력해준다
			}
		}
		printf("\n"); // 한 문자열 다 반복하면 개행
	}
}
```
